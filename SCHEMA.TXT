CREATE TABLE Tara(
    CodTara number(10) NOT NULL,
    Nume varchar2(128) NOT NULL,
    CONSTRAINT tara_pk PRIMARY KEY(CodTara)
);

CREATE TABLE Oras(
    CodOras number(10) NOT NULL,
    Nume varchar2(128) NOT NULL,
    CodPostal number(10),
    CodTara number(10),
    CONSTRAINT oras_pk PRIMARY KEY(CodOras),
    CONSTRAINT tara_fk FOREIGN KEY(CodTara) REFERENCES Tara(CodTara)
);

CREATE TABLE Spital (
    CodSpital number(10) NOT NULL,
    Nume varchar2(128) NOT NULL,
    Strada varchar2(128),
    Numar varchar2(10),
    CodOras number(10),
    CodManager number(10),
    CONSTRAINT spital_pk PRIMARY KEY(CodSpital),
    CONSTRAINT oras_fk FOREIGN KEY(CodOras) REFERENCES Oras(CodOras)
);

CREATE TABLE Cladire(
    CodCladire number(10) NOT NULL,
    Nume varchar2(128),
    CodSpital number(10),
    CONSTRAINT cladire_pk PRIMARY KEY(CodCladire),
    CONSTRAINT spital_fk FOREIGN KEY(CodSpital) REFERENCES Spital(CodSpital)
);

CREATE TABLE Sectie_x_Cladire(
    CodCladire number(10) NOT NULL,
    CodSectie number(10) NOT NULL,
    CONSTRAINT sectie_x_cladire_pk PRIMARY KEY(CodCladire,CodSectie),
    CONSTRAINT sectie_x_cladire_1_fk FOREIGN KEY(CodCladire) REFERENCES Cladire(CodCladire),
    CONSTRAINT sectie_x_cladire_2_fk FOREIGN KEY(CodSectie) REFERENCES Sectie(CodSectie)
);

CREATE TABLE Sectie(
    CodSectie number(10) NOT NULL,
    Nume varchar2(128) NOT NULL,
    CONSTRAINT sectie_pk PRIMARY KEY(CodSectie)
);

CREATE TABLE Medicament(
    CodMedicament number(10) NOT NULL,
    Nume varchar2(128) NOT NULL,
    Descriere varchar2(255),
    Ingrediente varchar2(128),
    Contraindicatii varchar2(128),
    Lot number(10),
    DataExpirarii date,
    ModAdministrare varchar2(32),
    CONSTRAINT medicament_pk PRIMARY KEY(CodMedicament)
);

CREATE TABLE Reteta_x_Medicament(
    CodReteta number(10) NOT NULL,
    CodMedicament number(10) NOT NULL,
    Cantitate number(4,2),
    CONSTRAINT reteta_x_medicament_pk PRIMARY KEY(CodReteta,CodMedicament),
    CONSTRAINT reteta_x_medicament_1_fk FOREIGN KEY(CodReteta) REFERENCES Reteta(CodReteta),
    CONSTRAINT reteta_x_medicament_2_fk FOREIGN KEY(CodMedicament) REFERENCES Medicament(CodMedicament)
);

CREATE TABLE Reteta(
    CodReteta number(10) NOT NULL,
    Data date,
    CodConsultatie number(10),
    CONSTRAINT reteta_pk PRIMARY KEY(CodReteta)
);

CREATE TABLE Diagnostic(
    CodDiagnostic number(10) NOT NULL,
    Nume varchar2(128)NOT NULL,
    Descriere varchar2(128),
    CONSTRAINT diagnostic_pk PRIMARY KEY(CodDiagnostic)
);

CREATE TABLE TipIncapere(
    CodTipIncapere number(10) NOT NULL,
    Nume varchar2(128)NOT NULL,
    CONSTRAINT tip_incapere_pk PRIMARY KEY(CodTipIncapere)
);    

CREATE TABLE SalaOperatie(
    CodIncapere number(10) NOT NULL,
    Nume varchar2(128),
    CodSectie number(10) NOT NULL,
    Informatii varchar(255),
    CONSTRAINT sala_operatie_pk PRIMARY KEY(CodIncapere),
    CONSTRAINT sala_operatie_sectie_fk FOREIGN KEY(CodSectie) REFERENCES Sectie(CodSectie)
);

CREATE TABLE Cabinet(
    CodIncapere number(10) NOT NULL,
    Nume varchar2(128),
    CodSectie number(10) NOT NULL,
    Detalii varchar(255),
    CONSTRAINT cabinet_pk PRIMARY KEY(CodIncapere),
    CONSTRAINT cabinet_sectie_fk FOREIGN KEY(CodSectie) REFERENCES Sectie(CodSectie)
);

CREATE TABLE Salon(
    CodIncapere number(10) NOT NULL,
    Nume varchar2(128),
    CodSectie number(10) NOT NULL,
    Capacitate number(2),
    CONSTRAINT salon_pk PRIMARY KEY(CodIncapere),
    CONSTRAINT salon_sectie_fk FOREIGN KEY(CodSectie) REFERENCES Sectie(CodSectie)
);

CREATE TABLE Camera(
    CodIncapere number(10) NOT NULL,
    Nume varchar2(128),
    CodSectie number(10) NOT NULL,
    CodTipIncapere number(10),
    CONSTRAINT camera_pk PRIMARY KEY(CodIncapere),
    CONSTRAINT camera_sectie_fk FOREIGN KEY(CodSectie) REFERENCES Sectie(CodSectie),
    CONSTRAINT camera_tip_incapere_fk FOREIGN KEY(CodTipIncapere) REFERENCES TipIncapere(CodTipIncapere)
);

CREATE TABLE Doctor(
    CodAngajat number(10) NOT NULL,
    Nume varchar2(128) NOT NULL,
    Prenume varchar2(128),
    Sex varchar2(10),
    DataNasterii date,
    Salariu number(10,2),
    DataAngajarii date NOT NULL,
    DataFinala date NOT NULL,
    CodSectie number(10),
    Specializare varchar2(50),
    CodColegiulMedicilor varchar2(20),
    Grad varchar2(30),
    CodCabinet number(10),
    CONSTRAINT doctor_pk PRIMARY KEY(CodAngajat),
    CONSTRAINT doctor_sectie_fk FOREIGN KEY(CodSectie) REFERENCES Sectie(CodSectie),
    CONSTRAINT doctor_cabinet_fk FOREIGN KEY(CodCabinet) REFERENCES Cabinet(CodIncapere)
);

CREATE TABLE PersonalMedicalAsistente(
    CodAngajat number(10) NOT NULL,
    Nume varchar2(128) NOT NULL,
    Prenume varchar2(128),
    Sex varchar2(10),
    DataNasterii date,
    Salariu number(10,2),
    DataAngajarii date NOT NULL,
    DataFinala date NOT NULL,
    CodSectie number(10),
    Specializare varchar2(50),
    CONSTRAINT personal_medical_asistente_pk PRIMARY KEY(CodAngajat),
    CONSTRAINT personal_medical_asistente_sectie_fk FOREIGN KEY(CodSectie) REFERENCES Sectie(CodSectie)
);

CREATE TABLE PersonalSpital(
    CodAngajat number(10) NOT NULL,
    Nume varchar2(128) NOT NULL,
    Prenume varchar2(128),
    Sex varchar2(10),
    DataNasterii date,
    Salariu number(10,2),
    DataAngajarii date NOT NULL,
    DataFinala date NOT NULL,
    CodSectie number(10),
    Functie varchar2(50),
    CONSTRAINT personal_spital_pk PRIMARY KEY(CodAngajat),
    CONSTRAINT personal_spital_sectie_fk FOREIGN KEY(CodSectie) REFERENCES Sectie(CodSectie)
);

CREATE TABLE Pacient(
    CodPacient number(10) NOT NULL,
    Nume varchar2(128) NOT NULL,
    Prenume varchar2(128),
    CNP number(20),
    Serie_Numar varchar2(10),
    CONSTRAINT pacient_pk PRIMARY KEY(CodPacient)
);

CREATE TABLE FisaMedicala(
    CodFisaMedicala number(10) NOT NULL,
    Simptome varchar2(128),
    DataInternare date,
    DataExternare date,
    Observatii varchar2(128),
    CodPacient number(10),
    CONSTRAINT fisa_medicala_pk PRIMARY KEY(CodFisaMedicala),
    CONSTRAINT fisa_medicala_pacient_fk FOREIGN KEY(CodPacient) REFERENCES Pacient(CodPacient)
);


CREATE TABLE Consultatie(
    CodConsultatie number(10) NOT NULL,
    Data date,
    Observatii varchar2(255),
    CodFisaMedicala number(10),
    CodDiagnostic number(10),
    CodDoctor number(10),
    CONSTRAINT consultatie_pk PRIMARY KEY(CodConsultatie),
    CONSTRAINT consultatie_fisa_medicala_fk FOREIGN KEY(CodFisaMedicala) REFERENCES FisaMedicala(CodFisaMedicala),
    CONSTRAINT consultatie_diagnostic_fk FOREIGN KEY(CodDiagnostic) REFERENCES Diagnostic(CodDiagnostic),
    CONSTRAINT consultatie_doctor_fk FOREIGN KEY(CodDoctor) REFERENCES Doctor(CodAngajat)
);

CREATE TABLE Pacient_x_Salon(
    CodSalon number(10) NOT NULL,
    CodPacient number(10) NOT NULL,
    DataInternare date,
    DataExternare date,
    CONSTRAINT Pacient_x_Salon_pk PRIMARY KEY(CodPacient,CodSalon),
    CONSTRAINT Pacient_x_Salon_1_fk FOREIGN KEY(CodSalon) REFERENCES Salon(CodIncapere),
    CONSTRAINT Pacient_x_Salon_2_fk FOREIGN KEY(CodPacient) REFERENCES Pacient(CodPacient)
);

CREATE TABLE Spital_x_Doctor(
    CodSpital number(10) NOT NULL,
    CodAngajat number(10) NOT NULL,
    DataAngajare date,
    DataFinala date,
    CONSTRAINT Spital_x_Doctor_pk PRIMARY KEY(CodSpital,CodAngajat),
    CONSTRAINT Spital_x_Doctor_1_fk FOREIGN KEY(CodSpital) REFERENCES Spital(CodSpital),
    CONSTRAINT Spital_x_Doctor_2_fk FOREIGN KEY(CodAngajat) REFERENCES Doctor(CodAngajat)
);

CREATE TABLE Operatie(
    CodPacient number(10) NOT NULL,
    CodIncapere number(10) NOT NULL,
    CodDoctor number(10) NOT NULL,
    Data date,
    Observatii varchar2(255),
    CONSTRAINT operatie_pk PRIMARY KEY(CodPacient,CodIncapere,CodDoctor),
    CONSTRAINT operatie_1_fk FOREIGN KEY(CodPacient) REFERENCES Pacient(CodPacient),
    CONSTRAINT operatie_2_fk FOREIGN KEY(CodIncapere) REFERENCES SalaOperatie(CodIncapere),
    CONSTRAINT operatie_3_fk FOREIGN KEY(CodDoctor) REFERENCES Doctor(CodAngajat)
);
